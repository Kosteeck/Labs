#include "iostream"
using namespace std;

int main()
{
	setlocale(LC_ALL, "Russian");

	float meters;
	float arshins;
	float span;

	cout << "Пожалуйста, введите положительное количество метров:  ";
	cin >> meters;
	if (meters >= 0)
	{
		arshins = meters * 1.406;
		span = meters * 4.374;
		cout << "\n  Перевод вашего количества метров в аршины: " << arshins << endl;
		cout << "\n  Перевод вашего количества метров в пяди: " << span << endl;
	}
	else
	{
		cout << ""Я же просил положительное :(" << "\n\n";

	}
	system("pause");
	return 0;
}
